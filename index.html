from mrjob.job import MRJob

class MRChargeSummer(MRJob):
    def mapper(self, key, line):
        row = line.split(',')
        # Only process records for newborn (Diagnostic code) and from 2009.
        if row[108].strip() == '795' and row[-1].strip() == '2009':
            try:
                yield row[2], (int(row[31].strip()), 1)
            except ValueError:
                pass

    def reducer(self, hospital, charges_count):
        individual_charges, individual_counts = zip(*charges_count)
        total_charges = sum(individual_charges)
        count = sum(individual_counts)
        average_charge_per_visit = total_charges/count
        yield hospital, average_charge_per_visit

    def steps(self):
        return [self.mr(self.mapper, self.reducer),]

if __name__ == '__main__':
    MRChargeSummer.run()
